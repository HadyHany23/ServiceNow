# pharmacy

-- 1. ROLES TABLE (Admin can add/edit roles here)
CREATE TABLE roles (
id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
name text UNIQUE NOT NULL,
created_at timestamptz DEFAULT now()
);

-- 2. PROFILES TABLE (Connects Login Users to Roles)
CREATE TABLE profiles (
id uuid REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
full_name text,
role_id uuid REFERENCES roles(id) ON DELETE SET NULL,
updated_at timestamptz DEFAULT now()
);

-- 3. CATEGORIES TABLE
CREATE TABLE categories (
id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
name text UNIQUE NOT NULL
);

-- 4. MEDICINES TABLE (Inventory)
CREATE TABLE medicines (
id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
name text NOT NULL,
category text NOT NULL, -- You can change this to uuid REFERENCES categories(id) if you prefer IDs
price numeric NOT NULL,
stock integer DEFAULT 0,
expiry_date date,
created_at timestamptz DEFAULT now()
);

-- 5. CUSTOMERS TABLE (For Auto-fill by Phone)
CREATE TABLE customers (
id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
phone text UNIQUE NOT NULL,
name text,
address text,
created_at timestamptz DEFAULT now()
);

-- 6. ORDERS TABLE (The Receipt Header)
CREATE TABLE orders (
id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
customer_id uuid REFERENCES customers(id) ON DELETE SET NULL,
seller_id uuid REFERENCES profiles(id), -- Tracks which employee made the sale
total_price numeric NOT NULL,
created_at timestamptz DEFAULT now()
);

-- 7. ORDER ITEMS TABLE (The Cart/History Details)
CREATE TABLE order_items (
id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
order_id uuid REFERENCES orders(id) ON DELETE CASCADE,
medicine_id bigint REFERENCES medicines(id) ON DELETE SET NULL,
quantity integer NOT NULL,
unit_price numeric NOT NULL -- Saved at time of sale in case price changes later
);
